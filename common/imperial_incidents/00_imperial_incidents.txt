# default options are 0-based
# In console, can use imperial_incident [key] to immediately begin an incident

incident_pu_subject_joins_empire = {
	event = incident_generic.2
	default_option = 0
	
	can_stop = {
		OR = {
			NOT = {
				any_country = {
					has_country_flag = current_pu_incident_target_flag
				}
			}
			any_country = {
				has_country_flag = current_pu_incident_target_flag
				OR = {
					is_subject = no
					overlord = {
						is_part_of_hre = no
					}
				}
			}
		}
	}
	
	0 = { # join
		factor = 1
		modifier = {
			factor = 1000
			any_subject_country = {
				has_country_flag = current_pu_incident_target_flag
			}
		}
	}
	
	1 = { # denied
		factor = 1
	}
}

			}

		}
	}
}

incident_reformer_dissension = {
	event = reformer_dissension.1
	default_option = 1
	
	can_stop = {
		always = no
	}
	
	0 = { # stay protestant
		factor = 1
		modifier = {
			factor = 50
			religion = protestant
		}
	}
	
	1 = { # religion treaty
		factor = 1
		modifier = {
			factor = 50
			OR = {
				religion = catholic
				has_idea_group = humanist_ideas
			}
		}
	}
	
	2 = { # change hre religion to reformed
		factor = 1
		modifier = {
			factor = 100
			religion = reformed
		}
		modifier = {
			factor = 0
			NOT = {
				religion = reformed
			}
		}
	}
}
